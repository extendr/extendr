pub struct Symbol { /* private fields */ }

Wrapper for creating symbol objects.

use extendr_api::prelude::*;
test! {
    let chr = r!(Symbol::from_string("xyz"));
    assert_eq!(chr.as_symbol().unwrap().as_str(), "xyz");
}
