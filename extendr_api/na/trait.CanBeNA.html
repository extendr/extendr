impl CanBeNA for &str

Special “NA” string that represents null strings.

use extendr_api::prelude::*;
test! {
    assert_ne!(<&str>::na().as_ptr(), "NA".as_ptr());
    assert_eq!(<&str>::na(), "NA");
    assert_eq!("NA".is_na(), false);
    assert_eq!(<&str>::na().is_na(), true);
}
