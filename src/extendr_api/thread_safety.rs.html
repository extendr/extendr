1//! Provide limited protection for multithreaded access to the R API.
2use crate::*;
3use extendr_ffi::{
4    R_MakeUnwindCont, R_UnwindProtect, Rboolean, Rf_error, Rf_protect, Rf_unprotect,
5};
6use std::cell::Cell;
7use std::sync::Mutex;
8
9/// A global lock, that should represent the global lock on the R-API.
10/// It is not tied to an actual instance of R.
11static R_API_LOCK: Mutex<()> = Mutex::new(());
12
13thread_local! {
14    static THREAD_HAS_LOCK: Cell<bool> = const { Cell::new(false) };
15}
16
17/// Run `f` while ensuring that `f` runs in a single-threaded manner.
18///
19/// This is intended for single-threaded access of the R's C-API.
20/// It is possible to have nested calls of `single_threaded` without deadlocking.
21///
22/// Note: This will fail badly if the called function `f` panics or calls `Rf_error`.
23pub fn single_threaded<F, R>(f: F) -> R
24where
25    F: FnOnce() -> R,
26{
27    let has_lock = THREAD_HAS_LOCK.with(|x| x.get());
28
29    // acquire R-API lock
30    let _guard = if !has_lock {
31        Some(R_API_LOCK.lock().unwrap())
32    } else {
33        None
34    };
35
36    // this thread now has the lock
37    THREAD_HAS_LOCK.with(|x| x.set(true));
38
39    let result = f();
40
41    // release the R-API lock
42    if _guard.is_some() {
43        THREAD_HAS_LOCK.with(|x| x.set(false));
44    }
45
46    result
47}

49/// This function is used by the wrapper logic to catch
50/// panics on return.
51///
52#[doc(hidden)]
53pub fn handle_panic<F, R>(err_str: &str, f: F) -> R
54where
55    F: FnOnce() -> R,
56    F: std::panic::UnwindSafe,
57{
58    match std::panic::catch_unwind(f) {
59        Ok(res) => res,
60        Err(_) => {
61            let err_str = CString::new(err_str).unwrap();
62            unsafe { Rf_error(err_str.as_ptr()) }
63        }
64    }
65}

67static mut R_ERROR_BUF: Option<std::ffi::CString> = None;
68
69pub fn throw_r_error<S: AsRef<str>>(s: S) -> ! {
70    let s = s.as_ref();
71    unsafe {
72        R_ERROR_BUF = Some(std::ffi::CString::new(s).unwrap());
73        let ptr = std::ptr::addr_of!(R_ERROR_BUF);
74        Rf_error((*ptr).as_ref().unwrap().as_ptr());
75    };
76}

78/// Wrap an R function such as `Rf_findFunction` and convert errors and panics into results.
79/// ```ignore
80/// use extendr_api::prelude::*;
81/// test! {
82///    let res = catch_r_error(|| unsafe {
83///        throw_r_error("bad things!");
84///        std::ptr::null_mut()
85///    });
86///    assert_eq!(res.is_ok(), false);
87/// }
88/// ```
89pub fn catch_r_error<F>(f: F) -> Result<SEXP>
90where
91    F: FnOnce() -> SEXP + Copy,
92    F: std::panic::UnwindSafe,
93{
94    use std::os::raw;
95
96    unsafe extern "C" fn do_call<F>(data: *mut raw::c_void) -> SEXP
97    where
98        F: FnOnce() -> SEXP + Copy,
99    {
100        let data = data as *const ();
101        let f: &F = &*(data as *const F);
102        f()
103    }
104
105    unsafe extern "C" fn do_cleanup(_: *mut raw::c_void, jump: Rboolean) {
106        if jump != Rboolean::FALSE {
107            panic!("R has thrown an error.");
108        }
109    }
110
111    single_threaded(|| unsafe {
112        let fun_ptr = do_call::<F> as *const ();
113        let clean_ptr = do_cleanup as *const ();
114        let x = false;
115        let fun = std::mem::transmute::<
116            *const (),
117            Option<unsafe extern "C" fn(*mut std::ffi::c_void) -> *mut extendr_ffi::SEXPREC>,
118        >(fun_ptr);
119        let cleanfun = std::mem::transmute::<
120            *const (),
121            std::option::Option<unsafe extern "C" fn(*mut std::ffi::c_void, extendr_ffi::Rboolean)>,
122        >(clean_ptr);
123        let data = &f as *const _ as _;
124        let cleandata = &x as *const _ as _;
125        let cont = R_MakeUnwindCont();
126        Rf_protect(cont);
127
128        // Note that catch_unwind does not work for 32 bit windows targets.
129        let res = match std::panic::catch_unwind(|| {
130            R_UnwindProtect(fun, data, cleanfun, cleandata, cont)
131        }) {
132            Ok(res) => Ok(res),
133            Err(_) => Err("Error in protected R code".into()),
134        };
135        Rf_unprotect(1);
136        res
137    })
