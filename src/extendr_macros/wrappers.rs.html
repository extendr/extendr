1//! This is responsible for generating the C functions that act as wrappers of
2//! the exported Rust functions.
3//!
4//! extendr relies on the [`.Call`-interface](https://cran.r-project.org/doc/manuals/R-exts.html#Calling-_002eCall)
5//! In short, it is necessary the the signature of the C-function have [`SEXP`]
6//! as the type for return type, and argument types.
7//!
8//! For instance, if your function returns nothing, the return type is not
9//! allowed to be `void`, instead `SEXP` must be used, and one should return
10//! [`R_NilValue`].
11//!
12//! ## R wrappers
13//!
14//! Within R, you may call `rextendr::document()` to generate R functions,
15//! that use the `.Call`-interface, to call the wrapped Rust functions.
16//!
17//! You may also manually implement these wrappers, in order to do special
18//! type-checking, or other annotation, that could be more convenient to do
19//! on the R-side. The C-functions are named according to `"{WRAP_PREFIX}{prefix}{mod_name}"`.
20//! See [`WRAP_PREFIX`], and note that `prefix` is set specifically for methods in
21//! `extendr`-impl blocks, while for functions have no prefix.
22//!
23//! [`R_NilValue`]: https://extendr.github.io/libR-sys/libR_sys/static.R_NilValue.html
24//! [`SEXP`]: https://extendr.github.io/libR-sys/libR_sys/type.SEXP.html
25

33pub const META_PREFIX: &str = "meta__";
34pub const WRAP_PREFIX: &str = "wrap__";

36lazy_static::lazy_static! {
37    static ref STRUCT_DOCS: Mutex<HashMap<String, String>> = Mutex::new(HashMap::new());
38}

40/// Called by the struct‐level #[extendr] macro to register docstrings.
41pub fn register_struct_doc(name: &str, doc: &str) {
42    STRUCT_DOCS
43        .lock()
44        .unwrap()
45        .insert(name.to_string(), doc.to_string());
46}

48/// Retrieve the struct‐level docs (or empty if none).
49pub fn get_struct_doc(name: &str) -> String {
50    STRUCT_DOCS
51        .lock()
52        .unwrap()
53        .get(name)
54        .cloned()
55        .unwrap_or_default()
56}

58// Generate wrappers for a specific function.
59pub(crate) fn make_function_wrappers(
60    opts: &ExtendrOptions,
61    wrappers: &mut Vec<ItemFn>,
62    prefix: &str,
63    attrs: &[syn::Attribute],
64    sig: &mut syn::Signature,
65    self_ty: Option<&syn::Type>,
66) -> syn::Result<()> {
67    let rust_name = sig.ident.clone();
68
69    let r_name_str = if let Some(r_name) = opts.r_name.as_ref() {
70        r_name.clone()
71    } else {
72        sig.ident.to_string()
73    };
74
75    let mod_name = if let Some(mod_name) = opts.mod_name.as_ref() {
76        format_ident!("{}", mod_name)
77    } else {
78        sig.ident.clone()
79    };
80
81    let mod_name = sanitize_identifier(mod_name);
82    let wrap_name = format_ident!("{}{}{}", WRAP_PREFIX, prefix, mod_name);
83    let meta_name = format_ident!("{}{}{}", META_PREFIX, prefix, mod_name);
84
85    let rust_name_str = format!("{}", rust_name);
86    let c_name_str = format!("{}", mod_name);
87    let doc_string = get_doc_string(attrs);
88    let return_type_string = get_return_type(sig);
89    let opts_invisible = match opts.invisible {
90        Some(true) => quote!(Some(true)),
91        Some(false) => quote!(Some(false)),
92        None => quote!(None),
93    };
94
95    let inputs = &mut sig.inputs;
96    let has_self = matches!(inputs.iter().next(), Some(FnArg::Receiver(_)));
97
98    let call_name = if has_self {
99        let is_mut = match inputs.iter().next() {
100            Some(FnArg::Receiver(ref receiver)) => receiver.mutability.is_some(),
101            _ => false,
102        };
103        if is_mut {
104            // eg. Person::name(&mut self)
105            quote! { extendr_api::unwrap_or_throw_error(
106                <&mut #self_ty>::try_from(&mut _self_robj)
107            ).#rust_name }
108        } else {
109            // eg. Person::name(&self)
110            quote! { extendr_api::unwrap_or_throw_error(
111                <&#self_ty>::try_from(&_self_robj)
112            ).#rust_name }
113        }
114    } else if let Some(ref self_ty) = &self_ty {
115        // eg. Person::new()
116        quote! { <#self_ty>::#rust_name }
117    } else {
118        // eg. aux_func()
119        quote! { #rust_name }
120    };
121
122    // arguments for the wrapper with type being `SEXP`
123    let formal_args = inputs
124        .iter()
125        .map(|input| translate_formal(input, self_ty))
126        .collect::<syn::Result<Punctuated<FnArg, Token![,]>>>()?;
127
128    // extract the names of the arguments only (`mut` are ignored in `formal_args` already)
129    let sexp_args = formal_args
130        .clone()
131        .into_iter()
132        .map(|x| match x {
133            // the wrapper doesn't use `self` arguments
134            FnArg::Receiver(_) => unreachable!(),
135            FnArg::Typed(ref typed) => match typed.pat.as_ref() {
136                syn::Pat::Ident(ref pat_ident) => pat_ident.ident.clone(),
137                _ => unreachable!(),
138            },
139        })
140        .collect::<Vec<Ident>>();
141
142    // arguments from R (`SEXP`s) are converted to `Robj`
143    let convert_args: Vec<syn::Stmt> = inputs
144        .iter()
145        .map(translate_to_robj)
146        .collect::<syn::Result<Vec<syn::Stmt>>>()?;
147
148    let actual_args: Punctuated<Expr, Token![,]> =
149        inputs.iter().filter_map(translate_actual).collect();
150
151    let meta_args: Vec<Expr> = inputs
152        .iter_mut()
153        .map(|input| translate_meta_arg(input, self_ty))
154        .collect::<syn::Result<Vec<Expr>>>()?;
155    let len_meta_args = meta_args.len();
156
157    // Generate wrappers for rust functions to be called from R.
158    // Example:
159    // ```
160    // #[no_mangle]
161    // #[allow(non_snake_case)]
162    // pub extern "C" fn wrap__hello() -> extendr_api::SEXP {
163    //     unsafe {
164    //         use extendr_api::FromRobj;
165    //         extendr_api::Robj::from(hello()).get()
166    //     }
167    // }
168    // ```
169    let rng_start = if opts.use_rng {
170        {
171            quote!(single_threaded(|| unsafe {
172                extendr_api::GetRNGstate();
173            });)
174        }
175    } else {
176        Default::default()
177    };
178    let rng_end = if opts.use_rng {
179        {
180            quote!(single_threaded(|| unsafe {
181                extendr_api::PutRNGstate();
182            });)
183        }
184    } else {
185        Default::default()
186    };
187
188    // figure out if
189    // -> &Self
190    // -> &mut Self
191    // Or if instead of `Self` the type name is used directly
192    // -> &ImplType / &mut ImplType
193    let return_is_ref_self = {
194        match sig.output {
195            // matches -> () or no-return type
196            syn::ReturnType::Default => false,
197            // ignoring the `-> Self` or `-> ImplType`, as that is not a Reference-type
198            // matches -> &T or &mut T
199            syn::ReturnType::Type(_, ref return_type) => match return_type.as_ref() {
200                Type::Reference(ref reference_type) => {
201                    // checks if T is Self or explicit impl type name
202                    if let Type::Path(path) = reference_type.elem.as_ref() {
203                        let is_typename_impl_type = self_ty
204                            .map(|x| x == reference_type.elem.as_ref())
205                            .unwrap_or(false);
206                        path.path.is_ident("Self") || is_typename_impl_type
207                    } else {
208                        false
209                    }
210                }
211                _ => false,
212            },
213        }
214    };
215
216    let return_type_conversion = if return_is_ref_self {
217        // instead of converting &Self / &mut Self, pass on the passed
218        // ExternalPtr<Self>
219        quote!(
220            let return_ref_to_self = #call_name(#actual_args);
221
222            #(
223            let arg_ref = extendr_api::R_ExternalPtrAddr(#sexp_args)
224                .cast::<Box<dyn std::any::Any>>()
225                .as_ref()
226                .unwrap()
227                .downcast_ref::<#self_ty>()
228                .unwrap();
229            if std::ptr::addr_eq(
230                arg_ref,
231                std::ptr::from_ref(return_ref_to_self)) {
232                    return Ok(extendr_api::Robj::from_sexp(#sexp_args))
233                }
234            )*
235            Err(Error::ExpectedExternalPtrReference)
236        )
237    } else {
238        quote!(Ok(extendr_api::Robj::from(#call_name(#actual_args))))
239    };
240
241    // TODO: the unsafe in here is unnecessary
242    wrappers.push(parse_quote!(
243        #[no_mangle]
244        #[allow(non_snake_case, clippy::not_unsafe_ptr_arg_deref)]
245        pub extern "C" fn #wrap_name(#formal_args) -> extendr_api::SEXP {
246            use extendr_api::robj::*;
247
248            // pull RNG state before evaluation
249            #rng_start
250
251            let wrap_result_state: std::result::Result<
252                std::result::Result<extendr_api::Robj, extendr_api::Error>,
253                Box<dyn std::any::Any + Send>
254            > = unsafe {
255                std::panic::catch_unwind(std::panic::AssertUnwindSafe(move || -> std::result::Result<extendr_api::Robj, extendr_api::Error> {
256                    #(#convert_args)*
257                    #return_type_conversion
258                }))
259            };
260
261            // return RNG state back to r after evaluation
262            #rng_end
263
264            // any obj created in above unsafe scope, which are not moved into wrap_result_state are now dropped
265            match wrap_result_state {
266                Ok(Ok(zz)) => {
267                    return unsafe { zz.get() };
268                }
269                // any conversion error bubbled from #actual_args conversions of incoming args from R.
270                Ok(Err(conversion_err)) => {
271                    let err_string = conversion_err.to_string();
272                    drop(conversion_err); // try_from=true errors contain Robj, this must be dropped to not leak
273                    extendr_api::throw_r_error(&err_string);
274                }
275                // any panic (induced by user func code or if user func yields a Result-Err as return value)
276                Err(unwind_err) => {
277                    drop(unwind_err); //did not notice any difference if dropped or not.
278                    // It should be possible to downcast the unwind_err Any type to the error
279                    // included in panic. The advantage would be the panic cause could be included
280                    // in the R terminal error message and not only via std-err.
281                    // but it should be handled in a separate function and not in-lined here.
282                    let err_string = format!("User function panicked: {}", #r_name_str);
283                    // cannot use throw_r_error here for some reason.
284                    // handle_panic() exports err string differently than throw_r_error.
285                    extendr_api::handle_panic(err_string.as_str(), || panic!());
286                }
287            }
288            unreachable!("internal extendr error, this should never happen.")
289        }
290    ));
291
292    // Generate a function to push the metadata for a function.
293    wrappers.push(parse_quote!(
294        #[allow(non_snake_case)]
295        fn #meta_name(metadata: &mut Vec<extendr_api::metadata::Func>) {
296            let mut args = Vec::with_capacity(#len_meta_args);
297            #(
298                args.push(#meta_args);
299            )*
300            let args = args;
301
302            metadata.push(extendr_api::metadata::Func {
303                doc: #doc_string,
304                rust_name: #rust_name_str,
305                r_name: #r_name_str,
306                mod_name: #c_name_str,
307                args: args,
308                return_type: #return_type_string,
309                func_ptr: #wrap_name as * const u8,
310                hidden: false,
311                invisible: #opts_invisible,
312            })
313        }
314    ));
315
316    Ok(())
317}

319// Extract doc strings from attributes.
320pub fn get_doc_string(attrs: &[syn::Attribute]) -> String {
321    let mut res = String::new();
322    for attr in attrs {
323        if !attr.path().is_ident("doc") {
324            continue;
325        }
326
327        if let syn::Meta::NameValue(ref nv) = attr.meta {
328            if let Expr::Lit(ExprLit {
329                lit: syn::Lit::Str(ref litstr),
330                ..
331            }) = nv.value
332            {
333                if !res.is_empty() {
334                    res.push('\n');
335                }
336                res.push_str(&litstr.value());
337            }
338        }
339    }
340    res
341}

343pub fn get_return_type(sig: &syn::Signature) -> String {
344    match &sig.output {
345        syn::ReturnType::Default => "()".into(),
346        syn::ReturnType::Type(_, ref rettype) => type_name(rettype),
347    }

350pub fn mangled_type_name(type_: &Type) -> String {
351    let src = quote!( #type_ ).to_string();
352    let mut res = String::new();
353    for c in src.chars() {
354        if c != ' ' {
355            if c.is_alphanumeric() {
356                res.push(c)
357            } else {
358                let f = format!("_{:02x}", c as u32);
359                res.push_str(&f);
360            }
361        }
362    }
363    res
364}

366/// Return a simplified type name that will be meaningful to R. Defaults to a digest.
367// For example:
368// & Fred -> Fred
369// * Fred -> Fred
370// && Fred -> Fred
371// Fred<'a> -> Fred
372// &[i32] -> _hex_hex_hex_hex
373//
374pub fn type_name(type_: &Type) -> String {
375    match type_ {
376        Type::Path(syn::TypePath { path, .. }) => {
377            if let Some(ident) = path.get_ident() {
378                ident.to_string()
379            } else if path.segments.len() == 1 {
380                let seg = path.segments.clone().into_iter().next().unwrap();
381                seg.ident.to_string()
382            } else {
383                mangled_type_name(type_)
384            }
385        }
386        Type::Group(syn::TypeGroup { elem, .. }) => type_name(elem),
387        Type::Reference(syn::TypeReference { elem, .. }) => type_name(elem),
388        Type::Paren(syn::TypeParen { elem, .. }) => type_name(elem),
389        Type::Ptr(syn::TypePtr { elem, .. }) => type_name(elem),
390        _ => mangled_type_name(type_),
391    }
392}

394// Generate a list of arguments for the wrapper. All arguments are SEXP for .Call in R.
395pub fn translate_formal(input: &FnArg, self_ty: Option<&syn::Type>) -> syn::Result<FnArg> {
396    match input {
397        // function argument.
398        FnArg::Typed(ref pattype) => {
399            let pat = pattype.pat.as_ref();
400            // ensure that `mut` in args are ignored in the wrapper
401            let pat_ident = translate_only_alias(pat)?;
402            Ok(parse_quote! { #pat_ident: extendr_api::SEXP })
403        }
404        // &self / &mut self
405        FnArg::Receiver(ref receiver) => {
406            if !receiver.attrs.is_empty() || receiver.reference.is_none() {
407                return Err(syn::Error::new_spanned(
408                    input,
409                    "expected &self or &mut self",
410                ));
411            }
412            if self_ty.is_none() {
413                return Err(syn::Error::new_spanned(
414                    input,"found &self in non-impl function - have you missed the #[extendr] before the impl?"
415                ));
416            }
417            Ok(parse_quote! { _self : extendr_api::SEXP })
418        }
419    }
420}

422/// Returns only the alias from a function argument.
423///
424/// For example `mut x: Vec<i32>`, the alias is `x`, but the `mut` would still
425/// be present if only the `Ident` of `PatType` was used.
426fn translate_only_alias(pat: &syn::Pat) -> Result<&Ident, syn::Error> {
427    Ok(match pat {
428        syn::Pat::Ident(ref pat_ident) => &pat_ident.ident,
429        _ => {
430            return Err(syn::Error::new_spanned(
431                pat,
432                "failed to translate name of argument",
433            ));
434        }
435    })
436}

438// Generate code to make a metadata::Arg.
439fn translate_meta_arg(input: &mut FnArg, self_ty: Option<&syn::Type>) -> syn::Result<Expr> {
440    match input {
441        // function argument.
442        FnArg::Typed(ref mut pattype) => {
443            let pat = pattype.pat.as_ref();
444            let ty = pattype.ty.as_ref();
445            // here the argument name is extracted, without the `mut` keyword,
446            // ensuring the generated r-wrappers, can use these argument names
447            let pat_ident = translate_only_alias(pat)?;
448            let name_string = quote! { #pat_ident }.to_string();
449            let type_string = type_name(ty);
450            let default = if let Some(default) = get_defaults(&mut pattype.attrs) {
451                quote!(Some(#default))
452            } else if let Some(default) = get_named_lit(&mut pattype.attrs, "default") {
453                quote!(Some(#default))
454            } else {
455                quote!(None)
456            };
457            Ok(parse_quote! {
458                extendr_api::metadata::Arg {
459                    name: #name_string,
460                    arg_type: #type_string,
461                    default: #default
462                }
463            })
464        }
465        // &self
466        FnArg::Receiver(ref receiver) => {
467            if !receiver.attrs.is_empty() || receiver.reference.is_none() {
468                return Err(syn::Error::new_spanned(
469                    input,
470                    "expected &self or &mut self",
471                ));
472            }
473            if self_ty.is_none() {
474                return Err(syn::Error::new_spanned(
475                    input,
476            "found &self in non-impl function - have you missed the #[extendr] before the impl?"
477        )
478    );
479            }
480            let type_string = type_name(self_ty.unwrap());
481            Ok(parse_quote! {
482                extendr_api::metadata::Arg {
483                    name: "self",
484                    arg_type: #type_string,
485                    default: None
486                }
487            })
488        }
489    }
490}

492// Get defaults from #[extendr(default = "value")] attribute.
493fn get_defaults(attrs: &mut Vec<syn::Attribute>) -> Option<String> {
494    use syn::Lit;
495
496    let mut new_attrs = Vec::new();
497    let mut res = None;
498
499    for i in attrs.drain(0..) {
500        if let syn::Meta::List(ref meta_list) = i.meta {
501            if meta_list.path.is_ident("extendr") {
502                let mut default_value = None;
503                let mut theres_default = false;
504
505                let parse_result = meta_list.parse_nested_meta(|meta| {
506                    if meta.path.is_ident("default") {
507                        theres_default = true;
508                        let value = meta.value()?;
509                        if let Ok(Lit::Str(litstr)) = value.parse() {
510                            default_value = Some(litstr.value());
511                        }
512                    }
513                    Ok(())
514                });
515
516                if parse_result.is_ok() && theres_default {
517                    res = default_value;
518                    continue;
519                }
520            }
521        }
522
523        new_attrs.push(i);
524    }
525    *attrs = new_attrs;
526    res
527}

529/// Convert `SEXP` arguments into `Robj`.
530/// This maintains the lifetime of references.
531///
532/// These conversions are from R into Rust
533fn translate_to_robj(input: &FnArg) -> syn::Result<syn::Stmt> {
534    match input {
535        FnArg::Typed(ref pattype) => {
536            let pat = &pattype.pat.as_ref();
537            if let syn::Pat::Ident(ref ident) = pat {
538                let varname = format_ident!("_{}_robj", ident.ident);
539                let ident = &ident.ident;
540                // TODO: these do not need protection, as they come from R
541                Ok(parse_quote! { let #varname = extendr_api::robj::Robj::from_sexp(#ident); })
542            } else {
543                Err(syn::Error::new_spanned(
544                    input,
545                    "expect identifier as arg name",
546                ))
547            }
548        }
549        FnArg::Receiver(_) => {
550            // this is `mut`, in case of a mutable reference
551            Ok(parse_quote! { let mut _self_robj = extendr_api::robj::Robj::from_sexp(_self); })
552        }
553    }
554}

556// Generate actual argument list for the call (ie. a list of conversions).
557fn translate_actual(input: &FnArg) -> Option<Expr> {
558    match input {
559        FnArg::Typed(ref pattype) => {
560            let pat = &pattype.pat.as_ref();
561            if let syn::Pat::Ident(ref ident) = pat {
562                let varname = format_ident!("_{}_robj", ident.ident);
563                Some(parse_quote! {
564                    #varname.try_into()?
565                })
566            } else {
567                None
568            }
569        }
570        FnArg::Receiver(_) => {
571            // Do not use self explicitly as an actual arg.
572            None
573        }
574    }
575}

577// Get a single named literal from a list of attributes.
578// eg. #[default="xyz"]
579// Remove the attribute from the list.
580fn get_named_lit(attrs: &mut Vec<syn::Attribute>, name: &str) -> Option<String> {
581    let mut new_attrs = Vec::new();
582    let mut res = None;
583    for a in attrs.drain(0..) {
584        if let syn::Meta::NameValue(ref nv) = a.meta {
585            if nv.path.is_ident(name) {
586                if let Expr::Lit(ExprLit {
587                    lit: syn::Lit::Str(ref litstr),
588                    ..
589                }) = nv.value
590                {
591                    eprintln!("#[default = \"arg\"] is deprecated. Use #[extendr(default = \"arg\")] instead.");
592                    res = Some(litstr.value());
593                    continue;
594                }
595            }
596        }
597
598        new_attrs.push(a);
599    }
600    *attrs = new_attrs;
601    res
602}

604// Remove the raw identifier prefix (`r#`) from an [`Ident`]
605// If the `Ident` does not start with the prefix, it is returned as is.
606fn sanitize_identifier(ident: Ident) -> Ident {
607    static PREFIX: &str = "r#";
608    let (ident, span) = (ident.to_string(), ident.span());
609    let ident = match ident.strip_prefix(PREFIX) {
610        Some(ident) => ident.into(),
611        None => ident,
612    };
613
614    Ident::new(&ident, span)
615}
